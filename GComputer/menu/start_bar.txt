function init(computer)
    computer.StartBar_menu = SFUi.window(Vector(0, 0), Vector(200, 350), "")
    computer.StartBar_menu.draggable = false
    computer.StartBar_menu.visible = false
    
    computer.OpenStartMenu = SFUi.checkbox(computer.GComputer_windows, Vector(0,0), Vector(45,45), "", false, function(checked)
        computer.StartBar_menu.visible = checked
    end)

    computer.OpenStartMenu.add_draw = function(self)
        render.setColor(computer.main_color)
        computer.func.star(self.mins.x + self.size.x/2,self.mins.y + self.size.y/2,self.size.y-25,0)
    end

    computer.GComputer_gui:addComponent(computer.StartBar_menu)
end

function render_menu(computer,w,h)
    computer.OpenStartMenu.pos = Vector(1,h-45/2-7)
    computer.StartBar_menu.pos = Vector(0,h-350-45)
    
    if computer.OpenStartMenu.checked then
        computer.OpenStartMenu.hover = true
    end
end

return {init=init,render=render_menu}
